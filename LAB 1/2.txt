Table Airport {
  airport_id int [pk]
  name varchar
  country varchar
  state varchar
  city varchar
  created_at datetime
  updated_at datetime
}

Table Airline {
  airline_id int [pk]
  airline_code varchar
  name varchar
  country varchar
  created_at datetime
  updated_at datetime
}

Table Flight {
  flight_id int [pk]
  airline_id int [ref: > Airline.airline_id]
  departure_airport_id int [ref: > Airport.airport_id]
  arrival_airport_id int [ref: > Airport.airport_id]
  dep_gate varchar
  arr_gate varchar
  sched_dep datetime
  sched_arr datetime
  act_dep datetime
  act_arr datetime
  created_at datetime
  updated_at datetime
}

Table Passenger {
  passenger_id int [pk]
  first_name varchar
  last_name varchar
  gender varchar
  dob date
  citizenship_country varchar
  residence_country varchar
  passport_number varchar
  created_at datetime
  updated_at datetime
}

Table Booking {
  booking_id int [pk]
  flight_id int [ref: > Flight.flight_id]
  passenger_id int [ref: > Passenger.passenger_id]
  status varchar
  platform varchar
  ticket_price decimal
  created_at datetime
  updated_at datetime
}

Table Booking_Changes {
  change_id int [pk]
  booking_id int [ref: > Booking.booking_id]
  change_time datetime
  details text
}

Table Boarding_Pass {
  bp_id int [pk]
  booking_id int [ref: > Booking.booking_id]
  seat varchar
  boarding_time datetime
  created_at datetime
  updated_at datetime
}

Table Baggage {
  baggage_id int [pk]
  booking_id int [ref: > Booking.booking_id]
  weight_kg decimal
  created_at datetime
  updated_at datetime
}

Table Baggage_Check {
  check_id int [pk]
  baggage_id int [ref: > Baggage.baggage_id]
  passenger_id int [ref: > Passenger.passenger_id]
  result varchar
  created_at datetime
  updated_at datetime
}

Table Security_Check {
  sec_id int [pk]
  passenger_id int [ref: > Passenger.passenger_id]
  result varchar
  created_at datetime
  updated_at datetime
}
